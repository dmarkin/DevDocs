Решето Эратосфена

var arr = [],
    maxNumber = 100,
    p = 2;

for (var i = 2; i < maxNumber; i++) {
  arr[i] = true;
}

do {
  for (i = 2 * p; i < maxNumber; i += p) {
    arr[i] = false;
  }

  for (i = p + 1; i < maxNumber; i++) {
    if (arr[i]) break;
  }

  p = i;
} while (p * p < maxNumber);
